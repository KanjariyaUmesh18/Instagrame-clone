<script>
  const fi = document.getElementById('fInp');
  const pv = document.getElementById('preview');
  const dz = document.getElementById('dz');
  
  function handleFile(file) {
    if (file) {
      const reader = new FileReader();
      reader.onload = function(e) {
        pv.src = e.target.result;
        dz.classList.add('has-file');
      }
      reader.readAsDataURL(file);
    }
  }

  fi.addEventListener('change', function(e) {
    handleFile(e.target.files[0]);
  });

  // Drag & Drop
  dz.addEventListener('dragover', function(e) {
    e.preventDefault();
    dz.style.borderColor = '#b6845e';
    dz.style.background = 'rgba(245,232,220,0.7)';
  });
  dz.addEventListener('dragleave', function(e) {
    e.preventDefault();
    dz.style.borderColor = '';
    dz.style.background = '';
  });
  dz.addEventListener('drop', function(e) {
    e.preventDefault();
    dz.style.borderColor = '';
    dz.style.background = '';
    if (e.dataTransfer.files && e.dataTransfer.files[0]) {
      fi.files = e.dataTransfer.files;
      handleFile(e.dataTransfer.files[0]);
    }
  });

  // Mobile Menu
  (function() {
    var btn = document.getElementById('mobileMenuBtn');
    var sidebar = document.getElementById('mainSidebar');
    var overlay = document.getElementById('sidebarOverlay');
    function openSidebar() { sidebar.classList.add('mobile-open'); overlay.classList.add('open'); btn.setAttribute('aria-expanded','true'); btn.querySelector('i').className='fa-solid fa-xmark'; }
    function closeSidebar() { sidebar.classList.remove('mobile-open'); overlay.classList.remove('open'); btn.setAttribute('aria-expanded','false'); btn.querySelector('i').className='fa-solid fa-bars'; }
    if(btn) btn.addEventListener('click', function(){ sidebar.classList.contains('mobile-open')?closeSidebar():openSidebar(); });
    if(overlay) overlay.addEventListener('click', closeSidebar);
    document.addEventListener('keydown', function(e){ if(e.key==='Escape') closeSidebar(); });
  })();
</script>

    (function() {
      // ----- DOM elements -----
      const fileInput = document.getElementById('imageUpload');
      const previewContainer = document.getElementById('filePreviewContainer');
      const captionInput = document.getElementById('captionInput');
      const publishBtn = document.getElementById('publishBtn');
      const resetBtn = document.getElementById('resetBtn');
      const dropZone = document.getElementById('dropZone');

      // ----- helper: update file preview with animation -----
      function updateFilePreview(file) {
        if (file) {
          const fileName = file.name;
          const fileSize = (file.size / 1024).toFixed(1);
          previewContainer.innerHTML = `
            <div class="file-preview">
              <i class="fas fa-check-circle" style="color: #7f9f6b;"></i>
              <span><strong>${fileName}</strong>  ·  ${fileSize} KB</span>
              <i class="fas fa-times-circle" style="margin-left: 6px; color: #a0684a; cursor: pointer;" id="removeFileBtn"></i>
            </div>
          `;
          // remove file handler
          document.getElementById('removeFileBtn').addEventListener('click', function(e) {
            e.stopPropagation();
            fileInput.value = '';
            previewContainer.innerHTML = '';
            // reset dropzone look (optional)
            dropZone.style.borderColor = '#cfb6a2';
          });
          // subtle border change
          dropZone.style.borderColor = '#9f7f64';
        } else {
          previewContainer.innerHTML = '';
          dropZone.style.borderColor = '#cfb6a2';
        }
      }

      // ----- file input change -----
      fileInput.addEventListener('change', function(e) {
        const file = this.files[0];
        if (file) {
          updateFilePreview(file);
        } else {
          previewContainer.innerHTML = '';
        }
      });

      // ----- drag & drop visual (optional) -----
      dropZone.addEventListener('dragover', function(e) {
        e.preventDefault();
        this.style.background = 'rgba(225, 200, 175, 0.5)';
        this.style.borderColor = '#b8865f';
      });
      dropZone.addEventListener('dragleave', function(e) {
        e.preventDefault();
        this.style.background = 'rgba(245, 232, 220, 0.4)';
        this.style.borderColor = '#cfb6a2';
      });
      dropZone.addEventListener('drop', function(e) {
        e.preventDefault();
        this.style.background = 'rgba(245, 232, 220, 0.4)';
        this.style.borderColor = '#cfb6a2';
        const files = e.dataTransfer.files;
        if (files.length > 0) {
          fileInput.files = files;
          updateFilePreview(files[0]);
        }
      });

      // ----- reset button : clear caption & image -----
      resetBtn.addEventListener('click', function() {
        captionInput.value = '';
        fileInput.value = '';
        previewContainer.innerHTML = '';
        dropZone.style.borderColor = '#cfb6a2';
        // optional focus
        captionInput.focus();
      });

      // ----- publish button: toast-like feedback (animated) -----
      publishBtn.addEventListener('click', function(e) {
        e.preventDefault();
        const caption = captionInput.value.trim() || '(no caption)';
        const file = fileInput.files[0];
        const fileName = file ? file.name : '📎 no image selected';
        // create a temporary feedback message (with animation)

        toast.style.position = 'fixed';
        toast.style.bottom = '30px';
        toast.style.left = '50%';
        toast.style.transform = 'translateX(-50%)';
        toast.style.background = '#2d241e';
        toast.style.color = '#f5ede6';

        toast.style.zIndex = '999';
        toast.style.fontSize = '1rem';
        toast.style.letterSpacing = '0.2px';
        toast.style.border = '1px solid #bfa187';
        toast.innerHTML = `
          <i class="fas fa-check-circle" style="color: #b6c7a3; margin-right: 12px;"></i>
          story ready · “${caption}”
        `;
        document.body.appendChild(toast);

        // animate in
        toast.style.animation = 'floatIn 0.4s ease';

          e.target.closest('form').submit();

      });

      // ----- extra: if user clicks on drop zone, trigger file input (already does) -----

      // small initial preview if there's a file from default ? (none)
      // but we set a default caption only.
    })();

/* ── Save/Bookmark toggle (purely visual) ── */
function tgSave(btn) {
  var ic = btn.querySelector('i');
  if (btn.classList.contains('saved')) {
    btn.classList.remove('saved');
    ic.className = 'fa-regular fa-bookmark';
  } else {
    btn.classList.add('saved');
    ic.className = 'fa-solid fa-bookmark';
  }
}

/* ── Like button visual toggle (backend handled by href) ── */
function handleLike(e, btn, countId) {
  // Allow the href (backend like/unlike) to proceed — we just add visual feedback
  btn.classList.toggle('liked');
  btn.classList.add('pop');
  var ic = btn.querySelector('i');
  if (btn.classList.contains('liked')) {
    ic.className = 'fa-solid fa-heart';
  } else {
    ic.className = 'fa-regular fa-heart';
  }
  btn.addEventListener('animationend', function() {
    btn.classList.remove('pop');
  }, { once: true });
  // Note: we do NOT call e.preventDefault() — the link navigates to toggle like on backend
}

/* ── Comments peek toggle ── */
function toggleComments(sectionId, btn) {
  var section = document.getElementById(sectionId);
  if (!section) return;
  var isOpen = section.classList.contains('open');
  section.classList.toggle('open');
  btn.setAttribute('aria-expanded', !isOpen);
  btn.textContent = isOpen ? 'View all comments' : 'Hide comments';
}

/* ── Client-side comment preview (appends to list before form submit) ── */
function postCmt(sectionId, inputId, listId) {
  var inp = document.getElementById(inputId);
  var txt = inp.value.trim();
  if (!txt) return;
  var list = document.getElementById(listId);
  var d = document.createElement('div');
  d.className = 'comment-item';
  d.innerHTML = '<img class="comment-av" src="{{uid.profile_pic.url}}" alt="You">' +
    '<div class="comment-body"><span class="comment-name">You</span>' +
    '<div class="comment-text">' + txt.replace(/</g, '&lt;') + '</div>' +
    '<div class="comment-time">Just now</div></div>';
  list.appendChild(d);
  list.scrollTop = list.scrollHeight;
  inp.value = '';
}

/* ── Mobile sidebar toggle ── */
(function() {
  var btn = document.getElementById('mobileMenuBtn');
  var sidebar = document.getElementById('mainSidebar');
  var overlay = document.getElementById('sidebarOverlay');

  function openSidebar() {
    sidebar.classList.add('mobile-open');
    overlay.classList.add('open');
    btn.setAttribute('aria-expanded', 'true');
    btn.querySelector('i').className = 'fa-solid fa-xmark';
  }
  function closeSidebar() {
    sidebar.classList.remove('mobile-open');
    overlay.classList.remove('open');
    btn.setAttribute('aria-expanded', 'false');
    btn.querySelector('i').className = 'fa-solid fa-bars';
  }

  if (btn) {
    btn.addEventListener('click', function() {
      if (sidebar.classList.contains('mobile-open')) {
        closeSidebar();
      } else {
        openSidebar();
      }
    });
  }
  if (overlay) {
    overlay.addEventListener('click', closeSidebar);
  }
  // Close sidebar on Escape
  document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape' && sidebar.classList.contains('mobile-open')) {
      closeSidebar();
    }
  });
})();
</script>

<script>
function togglePlay(btn){
  var s = btn.closest('.reel-slide');
  var v = s.querySelector('.reel-bg');
  var ic = btn.querySelector('i');
  if(v.paused){ v.play(); ic.className='fa-solid fa-pause'; btn.style.opacity='0'; }
  else { v.pause(); ic.className='fa-solid fa-play'; btn.style.opacity='1'; }
}
var cont = document.getElementById('reelsContainer');
cont.addEventListener('scroll', function() {
  var slides = document.querySelectorAll('.reel-slide');
  var cy = cont.scrollTop + cont.clientHeight/2;
  slides.forEach(function(s){
    var top = s.offsetTop;
    var bot = top + s.offsetHeight;
    var v = s.querySelector('.reel-bg');
    var ic = s.querySelector('.reel-play-center i');
    var btn = s.querySelector('.reel-play-center');
    if(cy >= top && cy <= bot){
      if(v.paused) { v.play(); ic.className='fa-solid fa-pause'; btn.style.opacity='0'; }
    } else {
      if(!v.paused) { v.pause(); ic.className='fa-solid fa-play'; btn.style.opacity='1'; }
    }
  });
});
(function() {
  var btn = document.getElementById('mobileMenuBtn');
  var sidebar = document.getElementById('mainSidebar');
  var overlay = document.getElementById('sidebarOverlay');
  function openSidebar() { sidebar.classList.add('mobile-open'); overlay.classList.add('open'); btn.setAttribute('aria-expanded','true'); btn.querySelector('i').className='fa-solid fa-xmark'; }
  function closeSidebar() { sidebar.classList.remove('mobile-open'); overlay.classList.remove('open'); btn.setAttribute('aria-expanded','false'); btn.querySelector('i').className='fa-solid fa-bars'; }
  if(btn) btn.addEventListener('click', function(){ sidebar.classList.contains('mobile-open')?closeSidebar():openSidebar(); });
  if(overlay) overlay.addEventListener('click', closeSidebar);
  document.addEventListener('keydown', function(e){ if(e.key==='Escape') closeSidebar(); });
})();
</script>
